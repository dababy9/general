# groupby() returns a iterable object of dataframes, grouped by having the same 'BUYER_COUNTY' value
# ["DOSAGE_UNIT"] specifies only the pill count to be looked at
# sum() just sums up the pill counts for each order, for each county, and returns a Series with the keys as counties and the values as total pill counts
# sort_values(ascending=False) sorts the series by total pill counts (ascending to have the largest pill counts at the top)
# head(5) grabs the top 5 rows of the Series
# index.values just grabs the keys of the top 5 rows and then returns the array of keys
print(df.groupby('BUYER_COUNTY')["DOSAGE_UNIT"].sum().sort_values(ascending=False).head(5).index.values)

['GREENVILLE' 'SPARTANBURG' 'HORRY' 'CHARLESTON' 'FLORENCE']

4. Which five counties bought the most pills per capita?¶

In [6]:
# read county populations into DataFrame
pop = pd.read_csv('datasets/populations.csv')

# extract DataFrame with counties and total pill counts
pills = df.groupby('BUYER_COUNTY')["DOSAGE_UNIT"].sum().to_frame().reset_index()

# merge new population Series with grouped DataFrame
merged = pd.merge(pop, pills, on='BUYER_COUNTY')

# create new column by dividing pill count by population (pills per capita)
merged['PILLS_PER_CAPITA'] = merged['DOSAGE_UNIT'] / merged['population']

# return top 5 counties sorted by the new column
print(merged.sort_values(by='PILLS_PER_CAPITA', ascending=False)['BUYER_COUNTY'].head(5).values)

['DARLINGTON' 'FLORENCE' 'OCONEE' 'DILLON' 'CHEROKEE']
